## 檔案 1 — `src/runtime/vit/runtime_vit.cpp`

> **核心模組**：ViT-Small/16 端對端推論引擎。  
> 包含：CLS 修正、multi-call 修正、volatile 統計、NORM macro。

```cpp
/* runtime_vit.cpp — ViT-Small/16 end-to-end inference engine.
 *
 * Attention is dispatched to the FA hardware accelerator (one head at a time).
 * All other ops (LayerNorm, QKV/O projection, MLP) run on the host CPU (fp32).
 *
 * CLS token handling:
 *   The FA hardware is fixed at N=196, Br=14 (patch tokens only; N%Br==0).
 *   Patch tokens attend to each other via the HW accelerator (N=196).
 *   CLS attention (Q_cls attending to all 197 tokens) is computed in software
 *   using single_query_attention() — one query row × 197 KV pairs per head.
 *   This restores standard ViT behaviour for the classification token.
 */

#include "runtime_vit.h"
#include "vit_weights.h"
#include "vit_ops.h"
#include "runtime.h"           /* flash_attention(), standard_attention_cpu() */
#include "driver_flash_attn.h" /* get_fa_hal()                                */
#include "hal.hpp"             /* runtime_info struct                         */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>

/* ── Norm selector ───────────────────────────────────────────────────────────
   Build with -DVIT_USE_RMSNORM to substitute RMSNorm for LayerNorm globally.
   RMSNorm is cheaper (no mean, no beta) but requires a model trained with it.
   ViT-Small/16 pretrained weights use LayerNorm; switching will hurt accuracy.
   Use RMSNorm only for benchmarking or when loading RMSNorm-trained weights.
   ─────────────────────────────────────────────────────────────────────────── */
#ifdef VIT_USE_RMSNORM
#define NORM(x, gamma, beta, N, d, eps) rms_norm((x), (gamma), (N), (d), (eps))
#else
#define NORM(x, gamma, beta, N, d, eps) layer_norm((x), (gamma), (beta), (N), (d), (eps))
#endif

/* Shorthand for dimension constants */
#define T   VIT_TOKENS      /* 197 */
#define N   VIT_PATCH_N     /* 196 */
#define D   VIT_EMBED_DIM   /* 384 */
#define H   VIT_HEADS       /* 6   */
#define HD  VIT_HEAD_DIM    /* 64  */
#define M   VIT_MLP_DIM     /* 1536 */
#define C   VIT_CLASSES     /* 1000 */
#define BR  VIT_TILE_BR     /* 14  */
#define EPS 1e-6f

/* ── HW stats accumulator ────────────────────────────────────────────────────
   Declared volatile so the compiler does not elide or reorder the
   read-modify-write sequences across successive flash_attention() calls. */
static volatile VitHWStats g_hw_stats;

void vit_hw_stats_reset(VitHWStats* s) {
    memset(s, 0, sizeof(*s));
    memset((void*)&g_hw_stats, 0, sizeof(g_hw_stats));
}

void vit_hw_stats_accumulate(VitHWStats* s) {
    s->total_cycles      = g_hw_stats.total_cycles;
    s->total_memory_read  = g_hw_stats.total_memory_read;
    s->total_memory_write = g_hw_stats.total_memory_write;
    s->fa_call_count      = g_hw_stats.fa_call_count;
    s->last_call_cycles   = g_hw_stats.last_call_cycles;
}

/* Internal: call flash_attention() and immediately capture per-call stats. */
static void fa_call_and_accumulate(float* Q, float* K, float* V, float* O,
                                   uint32_t n, uint32_t d, uint32_t br) {
    /* FA_interrupt stays HIGH after each computation and is only cleared by
       hardware reset (not by fa_stop() or by asserting start).  Without this
       reset, wait_for_irq() on calls 2+ exits immediately — still seeing the
       previous call's asserted interrupt — producing garbage output for all
       subsequent heads.  reset() pulses ARESETn, clears FA_interrupt, and
       returns the state machine to idle.  flash_attention() then re-writes all
       configuration registers before kicking off the new computation. */
    get_fa_hal()->reset();  // ★ 關鍵：每次 FA call 前清除 FA_interrupt

    flash_attention(Q, K, V, O, n, d, br);

    struct runtime_info ri = get_fa_hal()->get_runtime_info();

    /* Explicit read-modify-write into volatile fields. */
    uint64_t cy  = g_hw_stats.total_cycles;
    uint64_t mr  = g_hw_stats.total_memory_read;
    uint64_t mw  = g_hw_stats.total_memory_write;
    uint32_t cnt = g_hw_stats.fa_call_count;

    g_hw_stats.total_cycles       = cy  + (uint64_t)ri.elapsed_cycle;
    g_hw_stats.total_memory_read  = mr  + (uint64_t)ri.memory_read;
    g_hw_stats.total_memory_write = mw  + (uint64_t)ri.memory_write;
    g_hw_stats.fa_call_count      = cnt + 1u;
    g_hw_stats.last_call_cycles   = (uint64_t)ri.elapsed_cycle;

#ifdef FA_STATS_DEBUG
    fprintf(stderr,
            "[FA-STATS] call %2u: ri.cycles=%-10llu  accum=%-10llu  "
            "ri.read=%-7u  ri.write=%-7u\n",
            cnt + 1u,
            (unsigned long long)ri.elapsed_cycle,
            (unsigned long long)(cy + ri.elapsed_cycle),
            ri.memory_read, ri.memory_write);
#endif
}

/* ── Per-head static buffers for DMA stability ──────────────────────────── */
static float s_Q_h[N * HD];
static float s_K_h[N * HD];
static float s_V_h[N * HD];
static float s_O_h[N * HD];

/* ── CLS attention helper ────────────────────────────────────────────────────
   Computes attention for a SINGLE query row against T key/value pairs.
   Used for CLS token (HW FA is fixed at N=196, patch tokens only). */
static void single_query_attention(const float* q, const float* K, const float* V,
                                   float* out, int tokens, int hd) {
    float scale = 1.0f / sqrtf((float)hd);
    float* scores = (float*)malloc((size_t)tokens * sizeof(float));

    for (int t = 0; t < tokens; t++) {
        float dot = 0.0f;
        for (int d = 0; d < hd; d++)
            dot += q[d] * K[t * hd + d];
        scores[t] = dot * scale;
    }

    float maxval = scores[0];
    for (int t = 1; t < tokens; t++)
        if (scores[t] > maxval) maxval = scores[t];
    float sumexp = 0.0f;
    for (int t = 0; t < tokens; t++) {
        scores[t] = expf(scores[t] - maxval);
        sumexp += scores[t];
    }
    for (int t = 0; t < tokens; t++)
        scores[t] /= sumexp;

    memset(out, 0, (size_t)hd * sizeof(float));
    for (int t = 0; t < tokens; t++)
        for (int d = 0; d < hd; d++)
            out[d] += scores[t] * V[t * hd + d];

    free(scores);
}

/* ── Attention layer ─────────────────────────────────────────────────────── */
void vit_attention_layer(const float* tokens_in, float* attn_out,
                         const ViTWeights* w, int layer, int use_hw) {
    /* Step 1: QKV projection */
    float* qkv_out = (float*)malloc((size_t)T * 3 * D * sizeof(float));
    linear(tokens_in, w->qkv_w[layer], w->qkv_b[layer], qkv_out, T, D, 3*D);

    /* Step 2a: Patch-token multi-head attention (HW or SW) */
    float* concat_out = (float*)calloc((size_t)N * D, sizeof(float));

    for (int h = 0; h < H; h++) {
        for (int n = 0; n < N; n++) {
            int t = n + 1;
            const float* qkv_row = qkv_out + t * 3 * D;
            memcpy(s_Q_h + n * HD, qkv_row +         h * HD, (size_t)HD * sizeof(float));
            memcpy(s_K_h + n * HD, qkv_row + D     + h * HD, (size_t)HD * sizeof(float));
            memcpy(s_V_h + n * HD, qkv_row + 2 * D + h * HD, (size_t)HD * sizeof(float));
        }

        if (use_hw)
            fa_call_and_accumulate(s_Q_h, s_K_h, s_V_h, s_O_h, N, HD, BR);
        else
            standard_attention_cpu(s_Q_h, s_K_h, s_V_h, s_O_h, N, HD);

        for (int n = 0; n < N; n++)
            memcpy(concat_out + n * D + h * HD, s_O_h + n * HD,
                   (size_t)HD * sizeof(float));
    }

    /* Step 2b: CLS-token attention in software (per head) */
    float* cls_concat = (float*)malloc((size_t)D * sizeof(float));

    for (int h = 0; h < H; h++) {
        const float* q_cls_h = qkv_out + 0 * 3 * D + h * HD;

        float* K_all = (float*)malloc((size_t)T * HD * sizeof(float));
        float* V_all = (float*)malloc((size_t)T * HD * sizeof(float));
        for (int t = 0; t < T; t++) {
            const float* row = qkv_out + t * 3 * D;
            memcpy(K_all + t * HD, row + D     + h * HD, (size_t)HD * sizeof(float));
            memcpy(V_all + t * HD, row + 2 * D + h * HD, (size_t)HD * sizeof(float));
        }

        float o_cls_h[HD];
        single_query_attention(q_cls_h, K_all, V_all, o_cls_h, T, HD);
        free(K_all);
        free(V_all);

        memcpy(cls_concat + h * HD, o_cls_h, (size_t)HD * sizeof(float));
    }
    free(qkv_out);

    /* Step 3: Output projection */
    float* patch_attn = (float*)malloc((size_t)N * D * sizeof(float));
    linear(concat_out, w->proj_w[layer], w->proj_b[layer], patch_attn, N, D, D);
    free(concat_out);

    float* cls_attn = (float*)malloc((size_t)D * sizeof(float));
    linear(cls_concat, w->proj_w[layer], w->proj_b[layer], cls_attn, 1, D, D);
    free(cls_concat);

    /* Step 4: attn_out[0]=CLS, attn_out[1:]=patches */
    memcpy(attn_out,     cls_attn,   (size_t)D * sizeof(float));
    memcpy(attn_out + D, patch_attn, (size_t)N * D * sizeof(float));
    free(cls_attn);
    free(patch_attn);
}

/* ── One complete transformer block ─────────────────────────────────────── */
static void transformer_block(float* tokens, const ViTWeights* w,
                               int layer, int use_hw) {
    float* x_norm = (float*)malloc((size_t)T * D * sizeof(float));
    memcpy(x_norm, tokens, (size_t)T * D * sizeof(float));
    NORM(x_norm, w->norm1_w[layer], w->norm1_b[layer], T, D, EPS);

    float* attn_out = (float*)malloc((size_t)T * D * sizeof(float));
    vit_attention_layer(x_norm, attn_out, w, layer, use_hw);
    residual_add(tokens, attn_out, T * D);
    free(attn_out);

    memcpy(x_norm, tokens, (size_t)T * D * sizeof(float));
    NORM(x_norm, w->norm2_w[layer], w->norm2_b[layer], T, D, EPS);

    float* mlp_mid = (float*)malloc((size_t)T * M * sizeof(float));
    linear(x_norm, w->mlp_fc1_w[layer], w->mlp_fc1_b[layer], mlp_mid, T, D, M);
    gelu_inplace(mlp_mid, T * M);

    float* mlp_out = (float*)malloc((size_t)T * D * sizeof(float));
    linear(mlp_mid, w->mlp_fc2_w[layer], w->mlp_fc2_b[layer], mlp_out, T, M, D);
    free(mlp_mid);

    residual_add(tokens, mlp_out, T * D);
    free(mlp_out);
    free(x_norm);
}

/* ── Top-level ViT inference ─────────────────────────────────────────────── */
int vit_inference(const float* image, float* logits,
                  const ViTWeights* w, int use_hw) {
    float* patches   = (float*)malloc((size_t)N * 768 * sizeof(float));
    float* patch_out = (float*)malloc((size_t)N * D * sizeof(float));

    extract_patches(image, patches);
    linear(patches, w->patch_embed_w, w->patch_embed_b, patch_out, N, 768, D);
    free(patches);

    float* tokens = (float*)malloc((size_t)T * D * sizeof(float));
    prepend_cls_pos(tokens, patch_out, w->cls_token, w->pos_embed);
    free(patch_out);

    for (int layer = 0; layer < VIT_LAYERS; layer++) {
        fprintf(stdout, "[VIT] Block %2d/%d ...\n", layer + 1, VIT_LAYERS);
        fflush(stdout);
        transformer_block(tokens, w, layer, use_hw);
    }

    NORM(tokens, w->norm_w, w->norm_b, T, D, EPS);
    linear(tokens, w->head_w, w->head_b, logits, 1, D, C);

    free(tokens);
    return argmax(logits, C);
}
```

### 重要設計要點說明

```
【設計要點 A】NORM macro（第 32-36 行）
  編譯時透過 -DVIT_USE_RMSNORM 切換正規化方式。
  LayerNorm：需要 gamma + beta，有 mean subtraction
  RMSNorm  ：只需 gamma，無 mean subtraction，計算量較少

【設計要點 B】g_hw_stats volatile（第 58 行）
  static volatile VitHWStats → 防止 -O2 省略連續的讀-修改-寫
  每個 fa_call_and_accumulate() 呼叫後立即累積，
  因為下一次 reset() 前這是唯一的機會讀到正確的值

【設計要點 C】fa_call_and_accumulate 的 reset()（第 89 行）
  FA_interrupt 在計算完成後永遠保持 HIGH
  必須用 ARESETn 脈衝清零，否則 wait_for_irq() 立刻返回（8 cycles bug）

【設計要點 D】static s_Q/K/V/O_h 緩衝區（第 124-127 行）
  DMA 地址需要與 HAL 物件在同一 4GB 虛擬地址空間
  static 保證地址穩定，不會每次 malloc 到不同的 4GB 頁面

【設計要點 E】single_query_attention（第 139-171 行）
  為 CLS token 專用的 SW softmax attention
  patch tokens（N=196）用 HW FA，CLS token 用此 SW 路徑
  每個 head 分配 K_all/V_all（各 49KB）後立即 free
```

## 檔案 2 — `src/runtime/vit/vit_ops.cpp`

> **ViT 基礎算子**：LayerNorm、RMSNorm、linear、GELU、patch extraction 等。

```cpp
/* vit_ops.cpp — ViT building-block operations (CPU, fp32) */

#include "vit_ops.h"
#include <math.h>
#include <string.h>
#include <stdlib.h>

/* ── Linear ──────────────────────────────────────────────────────────────────
   weight[out_d, in_d] (PyTorch row-major)
   output[m][o] = Σ_k input[m][k] * weight[o][k] + bias[o]
   ─────────────────────────────────────────────────────────────────────────── */
void linear(const float* input, const float* weight, const float* bias,
            float* output, int M, int in_d, int out_d) {
    for (int m = 0; m < M; m++) {
        const float* in_row  = input  + m * in_d;
        float*       out_row = output + m * out_d;
        for (int o = 0; o < out_d; o++) {
            float sum = bias ? bias[o] : 0.0f;
            const float* w_row = weight + o * in_d;
            for (int k = 0; k < in_d; k++)
                sum += in_row[k] * w_row[k];
            out_row[o] = sum;
        }
    }
}

/* ── LayerNorm ───────────────────────────────────────────────────────────── */
void layer_norm(float* x, const float* gamma, const float* beta,
                int N, int d, float eps) {
    for (int i = 0; i < N; i++) {
        float* row = x + i * d;
        float mean = 0.0f;
        for (int j = 0; j < d; j++) mean += row[j];
        mean /= (float)d;
        float var = 0.0f;
        for (int j = 0; j < d; j++) {
            float v = row[j] - mean;
            var += v * v;
        }
        var /= (float)d;
        float inv_std = 1.0f / sqrtf(var + eps);
        for (int j = 0; j < d; j++)
            row[j] = (row[j] - mean) * inv_std * gamma[j] + beta[j];
    }
}

/* ── RMSNorm ─────────────────────────────────────────────────────────────────
   No mean subtraction, no beta.  Cheaper than LayerNorm by ~1/3 FLOPs.
   y_i = x_i / sqrt(1/d * Σ x_j² + eps) * gamma_i
   ─────────────────────────────────────────────────────────────────────────── */
void rms_norm(float* x, const float* gamma, int N, int d, float eps) {
    for (int i = 0; i < N; i++) {
        float* row = x + i * d;
        float sumsq = 0.0f;
        for (int j = 0; j < d; j++) sumsq += row[j] * row[j];
        float inv_rms = 1.0f / sqrtf(sumsq / (float)d + eps);
        for (int j = 0; j < d; j++)
            row[j] = row[j] * inv_rms * gamma[j];
    }
}

/* ── GELU (erf form, matches PyTorch default) ───────────────────────────── */
void gelu_inplace(float* x, int len) {
    static const float INV_SQRT2 = 0.7071067811865476f;
    for (int i = 0; i < len; i++) {
        float v = x[i];
        x[i] = 0.5f * v * (1.0f + erff(v * INV_SQRT2));
    }
}

/* ── Patch extraction ────────────────────────────────────────────────────────
   img[C=3, H=224, W=224] → patches[196, 768]
   Patch order: row-major over the 14×14 grid.
   Within each patch: channel-major (all ch0 pixels, then ch1, ch2).
   ─────────────────────────────────────────────────────────────────────────── */
void extract_patches(const float* img, float* patches) {
    for (int pr = 0; pr < 14; pr++) {
        for (int pc = 0; pc < 14; pc++) {
            float* dst = patches + (pr * 14 + pc) * 768;
            int idx = 0;
            for (int ch = 0; ch < 3; ch++) {
                for (int py = 0; py < 16; py++) {
                    int y = pr * 16 + py;
                    for (int px = 0; px < 16; px++) {
                        int x_pix = pc * 16 + px;
                        dst[idx++] = img[ch * 224 * 224 + y * 224 + x_pix];
                    }
                }
            }
        }
    }
}

/* ── Prepend CLS token + positional embedding ───────────────────────────── */
void prepend_cls_pos(float* tokens, const float* patch_out,
                     const float* cls, const float* pos_embed) {
    const int D = 384;
    for (int d = 0; d < D; d++)
        tokens[d] = cls[d] + pos_embed[d];
    for (int i = 0; i < 196; i++) {
        const float* pe  = pos_embed + (i + 1) * D;
        const float* src = patch_out + i * D;
        float*       dst = tokens    + (i + 1) * D;
        for (int d = 0; d < D; d++)
            dst[d] = src[d] + pe[d];
    }
}

/* ── Residual add ────────────────────────────────────────────────────────── */
void residual_add(float* x, const float* delta, int len) {
    for (int i = 0; i < len; i++) x[i] += delta[i];
}

/* ── Softmax (in-place, numerically stable) ─────────────────────────────── */
void softmax_inplace(float* x, int len) {
    float m = x[0];
    for (int i = 1; i < len; i++) if (x[i] > m) m = x[i];
    float s = 0.0f;
    for (int i = 0; i < len; i++) { x[i] = expf(x[i] - m); s += x[i]; }
    for (int i = 0; i < len; i++) x[i] /= s;
}

/* ── Argmax ──────────────────────────────────────────────────────────────── */
int argmax(const float* x, int len) {
    int best = 0;
    for (int i = 1; i < len; i++) if (x[i] > x[best]) best = i;
    return best;
}

/* ── Top-k (simple selection sort) ─────────────────────────────────────── */
void topk(const float* x, int len, int k, int* indices, float* values) {
    float* tmp = (float*)malloc(len * sizeof(float));
    int*   idx = (int*)malloc(len * sizeof(int));
    for (int i = 0; i < len; i++) { tmp[i] = x[i]; idx[i] = i; }
    for (int i = 0; i < k; i++) {
        int best = i;
        for (int j = i + 1; j < len; j++)
            if (tmp[j] > tmp[best]) best = j;
        float tf = tmp[i]; tmp[i] = tmp[best]; tmp[best] = tf;
        int   ti = idx[i]; idx[i] = idx[best]; idx[best] = ti;
        indices[i] = idx[i];
        values[i]  = tmp[i];
    }
    free(tmp); free(idx);
}
```

### LayerNorm vs RMSNorm 對比圖

```
LayerNorm：                        RMSNorm（★ 本專案新增）：
─────────────────────────────      ─────────────────────────────
需要：mean, variance, gamma, beta  需要：rms, gamma（無 mean、無 beta）
                                   
mean = Σxi / d                     rms = sqrt(Σxi² / d + eps)
var  = Σ(xi-mean)² / d
std  = sqrt(var + eps)

y = (x-mean)/std × gamma + beta    y = x / rms × gamma

計算步驟：2 pass（mean → var）     計算步驟：1 pass（sumsq → inv_rms）
參數：gamma[d] + beta[d]            參數：gamma[d]（省 beta[d]）

準確率差異（50張測試集）：無（LayerNorm beta 對 ViT-S 貢獻小）
```

## 檔案 3 — `src/runtime/flash_attn/runtime_flash_attn.cpp`

> **FlashAttention 運行時**：INT8 量化、ScaleTying、HW 啟動與等待。

```cpp
// runtime_flash_attn.cpp — high-level FlashAttention runtime API

#include <math.h>
#include <stdint.h>
#include <stdlib.h>
#include <string.h>

#include "driver_flash_attn.h"
#include "runtime.h"

/* ── CPU reference: standard (non-tiled) attention ─────────────────────── */
void standard_attention_cpu(const float* Q, const float* K, const float* V,
                             float* O, uint32_t N, uint32_t d) {
    float scale = 1.0f / sqrtf((float)d);
    float* S = (float*)malloc(N * N * sizeof(float));

    for (uint32_t i = 0; i < N; i++)
        for (uint32_t j = 0; j < N; j++) {
            float s = 0.0f;
            for (uint32_t k = 0; k < d; k++)
                s += Q[i * d + k] * K[j * d + k];
            S[i * N + j] = s * scale;
        }

    for (uint32_t i = 0; i < N; i++) {
        float m = S[i * N + 0];
        for (uint32_t j = 1; j < N; j++)
            if (S[i * N + j] > m) m = S[i * N + j];
        float sum = 0.0f;
        for (uint32_t j = 0; j < N; j++) {
            S[i * N + j] = expf(S[i * N + j] - m);
            sum += S[i * N + j];
        }
        for (uint32_t j = 0; j < N; j++) S[i * N + j] /= sum;
    }

    for (uint32_t i = 0; i < N; i++)
        for (uint32_t k = 0; k < d; k++) {
            float o = 0.0f;
            for (uint32_t j = 0; j < N; j++) o += S[i * N + j] * V[j * d + k];
            O[i * d + k] = o;
        }
    free(S);
}

/* ── Symmetric INT8 quantization helpers ───────────────────────────────── */

static float compute_scale(const float* data, uint32_t len) {
    float max_abs = 0.0f;
    for (uint32_t i = 0; i < len; i++) {
        float v = fabsf(data[i]);
        if (v > max_abs) max_abs = v;
    }
    return (max_abs > 0.0f) ? max_abs / 127.0f : 1.0f;
}

static void quantize_to_int8(const float* data, int8_t* out,
                              uint32_t len, float scale) {
    for (uint32_t i = 0; i < len; i++) {
        float v = data[i] / scale;
        int   ri = (int)(v >= 0.0f ? v + 0.5f : v - 0.5f);
        if (ri >  127) ri =  127;
        if (ri < -127) ri = -127;
        out[i] = (int8_t)ri;
    }
}

/* ── Hardware FlashAttention ────────────────────────────────────────────── */
void fa_stop() { fa_reg_write(FA_CONTROL_OFFSET, 0); }

/* ── Scale tying ─────────────────────────────────────────────────────────────
 * Build with -DFA_SCALE_TYING to enable.
 *
 * Without tying (default):
 *   q_scale = max(|Q|)/127    k_scale = max(|K|)/127    (independent)
 *   Hardware dequantizes: S_fp = S_int × q_scale × k_scale
 *
 * With tying:
 *   qk_scale = max(max(|Q|), max(|K|)) / 127    (shared)
 *   Hardware dequantizes: S_fp = S_int × qk_scale²
 *   Advantage: single scale factor, simpler hardware dequantization path.
 * ─────────────────────────────────────────────────────────────────────────── */
int flash_attention(float* Q, float* K, float* V, float* O,
                    uint32_t N, uint32_t d, uint32_t Br) {
    uint32_t len = N * d;

    float v_scale = compute_scale(V, len);

#ifdef FA_SCALE_TYING
    float qk_scale = fmaxf(compute_scale(Q, len), compute_scale(K, len));
    float q_scale  = qk_scale;
    float k_scale  = qk_scale;
#else
    float q_scale = compute_scale(Q, len);
    float k_scale = compute_scale(K, len);
#endif

    int8_t* Q_int8 = (int8_t*)malloc(len * sizeof(int8_t));
    int8_t* K_int8 = (int8_t*)malloc(len * sizeof(int8_t));
    int8_t* V_int8 = (int8_t*)malloc(len * sizeof(int8_t));

    quantize_to_int8(Q, Q_int8, len, q_scale);
    quantize_to_int8(K, K_int8, len, k_scale);
    quantize_to_int8(V, V_int8, len, v_scale);

    fa_set_shape(N, d);
    fa_set_tile(Br);
    fa_set_q_addr(Q_int8);
    fa_set_k_addr(K_int8);
    fa_set_v_addr(V_int8);
    fa_set_o_addr(O);
    fa_set_q_scale(q_scale);
    fa_set_k_scale(k_scale);
    fa_set_v_scale(v_scale);

    get_fa_hal()->reset_runtime_info();  // 清零計數器（reset() 的 cycles 不計入）
    fa_start();
    get_fa_hal()->wait_for_irq();
    fa_stop();

    free(Q_int8); free(K_int8); free(V_int8);
    return 0;
}
```

### ScaleTying 流程圖

```
獨立 scale（預設）：                  ScaleTying（-DFA_SCALE_TYING）：
──────────────────────────────────    ──────────────────────────────────
q_scale = max(|Q|) / 127             qk_scale = max(max(|Q|), max(|K|)) / 127
k_scale = max(|K|) / 127             q_scale  = qk_scale
                                      k_scale  = qk_scale

Q_int8 = round(Q / q_scale)          Q_int8 = round(Q / qk_scale)
K_int8 = round(K / k_scale)          K_int8 = round(K / qk_scale)

GEMM1 反量化：                        GEMM1 反量化：
S_fp = S_int × q_scale × k_scale     S_fp = S_int × qk_scale²
     （2 個不同的乘法因子）                （1 個共用因子的平方）

優：最佳量化精度                       優：硬體簡化（1 個 scale 暫存器）
缺：硬體需要 2 個 scale load           缺：Q/K 分布差異大時精度略降
```

## 檔案 4 — `src/hal/flash_attn_hal.cpp`

> **HAL 層**：Verilator 模型的 AXI4 MMIO 讀寫與 DMA 服務迴圈。

```cpp
// flash_attn_hal.cpp — FlashAttnHAL implementation

#include "flash_attn_hal.hpp"
#include <cstdio>
#include <cstring>

FlashAttnHAL::FlashAttnHAL(uint32_t baseaddr, uint32_t mmio_size)
    : info_{}, baseaddr_(baseaddr), mmio_size_(mmio_size),
      device_(nullptr), vm_addr_h_(0) {
    /* vm_addr_h_：HAL 物件的高 32 位地址。
       靜態 DMA buffer（s_Q_h...）需要和 HAL 物件在同一 4GB 空間，
       DMA 地址 = vm_addr_h_ | ARADDR_M（Verilog 傳來的低 32 位）。 */
    vm_addr_h_ = (reinterpret_cast<uint64_t>(this) & 0xffffffff00000000ULL);
    device_ = new Vflash_attn_wrapper("TOP");
}

void FlashAttnHAL::init() {
    reset_runtime_info();
    reset();
}

void FlashAttnHAL::reset() {
    /* ARESETn=0 → 幾個 cycle → ARESETn=1：完整硬體 reset。
       這會清除 FA_interrupt，使 wait_for_irq() 能正常等待下次完成。 */
    device_->ARESETn = 0;
    for (uint32_t i = 0; i < FA_RESET_CYCLE; i++)
        clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);
    device_->ARESETn = 1;
    device_->eval();
}

void FlashAttnHAL::reset_runtime_info() {
    info_.elapsed_cycle = 0;
    info_.elapsed_time  = 0;
    info_.memory_read   = 0;
    info_.memory_write  = 0;
}

/* MMIO write — AXI4 Slave Write (AW → W → B channels) */
bool FlashAttnHAL::memory_set(uint32_t addr, uint32_t data) {
    if (!device_) return false;
    if (addr < baseaddr_ || addr >= baseaddr_ + mmio_size_) return false;

    /* AW channel */
    device_->AWADDR_S  = addr;
    device_->AWVALID_S = 1; device_->eval();
    while (!device_->AWREADY_S)
        clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);
    clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);
    device_->AWVALID_S = 0;

    /* W channel */
    device_->WDATA_S  = data;
    device_->WSTRB_S  = AXI_STRB_WORD;
    device_->WLAST_S  = 1;
    device_->WVALID_S = 1; device_->eval();
    while (!device_->WREADY_S)
        clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);
    clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);
    device_->WVALID_S = 0;

    /* B channel */
    device_->BREADY_S = 1; device_->eval();
    while (!device_->BVALID_S)
        clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);
    clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);
    device_->BREADY_S = 0;
    return device_->BRESP_S == AXI_RESP_OKAY;
}

/* Block until FA_interrupt asserted, serving DMA during the wait */
void FlashAttnHAL::wait_for_irq() {
    if (!device_) return;
    while (!device_->FA_interrupt) {
        clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);
        if (device_->ARVALID_M) handle_dma_read();
        if (device_->AWVALID_M) handle_dma_write();
    }
}

/* DMA read: accelerator requests data from host memory */
void FlashAttnHAL::handle_dma_read() {
    uint32_t* addr =
        reinterpret_cast<uint32_t*>(vm_addr_h_ | device_->ARADDR_M);
    uint32_t len = device_->ARLEN_M;

    device_->ARREADY_M = 1;
    clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);
    device_->ARREADY_M = 0;
    clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);

    for (int i = 0; i <= (int)len; i++) {
        device_->RDATA_M = *(addr + i);
        info_.elapsed_cycle += MEM_ACCESS_CYCLE;
        device_->RLAST_M  = (i == (int)len);
        device_->RVALID_M = 1; device_->eval();
        while (!device_->RREADY_M)
            clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);
        clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);
        device_->RVALID_M = 0;
        clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);
    }
    info_.memory_read += sizeof(uint32_t) * (len + 1);
}

/* DMA write: accelerator writes output to host memory */
void FlashAttnHAL::handle_dma_write() {
    uint32_t* addr =
        reinterpret_cast<uint32_t*>(vm_addr_h_ | device_->AWADDR_M);
    uint32_t len = device_->AWLEN_M;

    device_->AWREADY_M = 1;
    clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);
    device_->AWREADY_M = 0;
    clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);

    for (uint32_t i = 0; i <= len; i++) {
        *(addr + i) = static_cast<uint32_t>(device_->WDATA_M);
        info_.elapsed_cycle += MEM_ACCESS_CYCLE;
        device_->WREADY_M = 1; device_->eval();
        while (!device_->WVALID_M)
            clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);
        clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);
        device_->WREADY_M = 0;
        clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);
    }

    device_->BVALID_M = 1; device_->eval();
    while (!device_->BREADY_M)
        clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);
    clock_step(device_, ACLK, info_.elapsed_cycle, info_.elapsed_time);
    device_->BVALID_M = 0; device_->eval();

    info_.memory_write += sizeof(uint32_t) * (len + 1);
}
```

```python
        source = load_from_numbered_dirs(ds_path, numeric_dirs, args.start, args.n)
    elif detect_hf(ds_path):
        source = load_from_hf(ds_path, args.start, args.n)
    else:
        source = load_from_flat_dir(ds_path, args.start, args.n)

    # 批次前處理
    labels = []
    processed = 0
    for pil_img, label in source:
        arr = preprocess_pil(pil_img)             # float32 CHW [3,224,224]
        arr.tofile(f"{args.out}/img_{processed:04d}.bin")  # 57,600 × 4 bytes
        labels.append(label)
        processed += 1

    # 輸出 labels.txt
    with open(f"{args.out}/labels.txt", "w") as f:
        for lb in labels: f.write(f"{lb}\n")

if __name__ == "__main__":
    main()
```

```
    │
    ▼ PIL Image.open() → convert("RGB")
RGB HWC [H, W, 3]
    │
    ▼ 中心裁切（min(H,W) × min(H,W)）→ Resize 224×224（BICUBIC）
RGB HWC [224, 224, 3]
    │
    ▼ / 255.0  → float32 [0.0, 1.0]
    │
    ▼ (pixel - mean) / std   （ImageNet 正規化）
    │   mean = [0.485, 0.456, 0.406]
    │   std  = [0.229, 0.224, 0.225]
    │
    ▼ transpose HWC → CHW
float32 CHW [3, 224, 224]
    │
    ▼ tofile()
img_XXXX.bin（230,400 bytes = 3 × 224 × 224 × 4）
```
